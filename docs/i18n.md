// app/[locale]/page.tsx
import { getDictionary } from '@/app/i18n/server';

export default async function Page({ params: { locale } }) {
const dict = await getDictionary(locale);

return (

<div>
<h1>{dict.home.title}</h1>
<p>{dict.home.description}</p>
</div>
);
}

export const defaultLocale = 'fr';
export const locales = ['fr', 'en', 'es', 'de'];

export type Locale = typeof locales[number];



## Bonnes Pratiques

### Nomenclature des Clés

- Utilisez des clés en anglais pour toutes les langues
- Organisez les clés par section/page
- Utilisez la notation `camelCase` pour les clés
- Évitez les clés trop longues ou trop imbriquées

### Paramètres Dynamiques

Pour les textes avec paramètres :

```json
// fr.json
{
  "user": {
    "greeting": "Bonjour, {{name}} !"
  }
}
```

```tsx
// Utilisation
<p>{t("user.greeting", { name: user.name })}</p>
```

### Pluralisation

Pour les textes avec pluralisation :

```json
// fr.json
{
  "notifications": {
    "count": "Vous avez {{count}} notification",
    "count_plural": "Vous avez {{count}} notifications"
  }
}
```

```tsx
// Utilisation
<p>{t("notifications.count", { count: notificationsCount })}</p>
```

### Dates et Nombres

Utilisez les fonctions de formatage adaptées à la locale :

```tsx
// Formatage de date
const formattedDate = new Intl.DateTimeFormat(locale).format(date);

// Formatage de nombre
const formattedNumber = new Intl.NumberFormat(locale).format(number);
```

## Navigation et Routes

Pour la navigation entre les pages tout en conservant la locale :

```tsx
// Utilisation de Link
import Link from "next/link";

<Link href={`/${locale}/profile`}>Profil</Link>;

// Utilisation du router
import { useRouter } from "next/navigation";

const router = useRouter();
router.push(`/${locale}/profile`);
```

## Détection de Langue

La détection de langue est gérée par le middleware Next.js dans `middleware.ts` :

- Vérification des paramètres de l'URL
- Vérification des cookies
- Vérification de l'en-tête Accept-Language
- Utilisation de la langue par défaut si aucune langue n'est détectée

## Ajout d'une Nouvelle Langue

1. Ajoutez le code de langue dans `app/i18n/settings.ts`
2. Créez un nouveau fichier de traduction dans `app/i18n/locales/`
3. Traduisez toutes les clés existantes
4. Ajoutez la langue dans les options du sélecteur de langue
